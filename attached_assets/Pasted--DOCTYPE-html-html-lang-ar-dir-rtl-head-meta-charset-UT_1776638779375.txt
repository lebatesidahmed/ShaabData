function pad(value) {
  return String(value).padStart(2, '0');
}

function updateCountdown() {
  const now = new Date();
  const diff = COUNTDOWN_TARGET - now;
  if (diff <= 0) {
    document.getElementById('cd-days').textContent = '00';
    document.getElementById('cd-hours').textContent = '00';
    document.getElementById('cd-mins').textContent = '00';
    document.getElementById('cd-secs').textContent = '🎉';
    return;
  }

  const days = Math.floor(diff / 86400000);
  const hours = Math.floor((diff % 86400000) / 3600000);
  const mins = Math.floor((diff % 3600000) / 60000);
  const secs = Math.floor((diff % 60000) / 1000);

  document.getElementById('cd-days').textContent = pad(days);
  document.getElementById('cd-hours').textContent = pad(hours);
  document.getElementById('cd-mins').textContent = pad(mins);
  document.getElementById('cd-secs').textContent = pad(secs);
}

function loadShoutouts() {
  try {
    const saved = window.localStorage.getItem(STORAGE_KEYS.shoutouts);
    return saved ? JSON.parse(saved) : defaultShoutouts;
  } catch (err) {
    return defaultShoutouts;
  }
}

function saveShoutouts(items) {
  try {
    window.localStorage.setItem(STORAGE_KEYS.shoutouts, JSON.stringify(items));
  } catch (err) {
    console.warn('Unable to save shoutouts', err);
  }
}

function renderShoutout(shoutout) {
  const emojis = ['🧠', '📊', '🤖', '🐍', '📈', '🔬', '💡', '🎯'];
  const avatar = emojis[Math.floor(Math.random() * emojis.length)];

    <div class="shoutout-item">
      <div class="shoutout-avatar">${avatar}</div>
      <div>
        <div class="shoutout-name">${shoutout.name.toUpperCase()}</div>
        <div class="shoutout-text">${shoutout.text}</div>
        <div class="shoutout-time">${shoutout.time}</div>
      </div>
    </div>`;
}

function renderShoutouts(items) {
  document.getElementById('shoutoutsFeed').innerHTML = items.map(renderShoutout).join('');
}

function getPollState() {
  try {
    return JSON.parse(window.localStorage.getItem(STORAGE_KEYS.poll));
  } catch (err) {
    return null;
  }
}

function savePollState() {
  try {
    window.localStorage.setItem(STORAGE_KEYS.poll, JSON.stringify({
      voted: pollData.voted,
      selected: pollData.selected,
      options: pollData.options,
    }));
  } catch (err) {
    console.warn('Unable to save poll state', err);
  }
}

function applyPollState() {
  const saved = getPollState();
  if (!saved) return;
  pollData.voted = Boolean(saved.voted);
  pollData.selected = saved.selected;
  if (Array.isArray(saved.options) && saved.options.length === pollData.options.length) {
    pollData.options = saved.options;
  }
}

function renderPoll() {
  const total = pollData.options.reduce((sum, option) => sum + option.votes, 0);
  document.getElementById('pollQuestion').textContent = pollData.question;
  const optEl = document.getElementById('pollOptions');
  optEl.innerHTML = pollData.options.map((opt, index) => {
    const pct = total ? Math.round((opt.votes / total) * 100) : 0;
    return `
      <button type="button" class="poll-option ${pollData.voted ? 'voted' : ''} ${pollData.selected === index ? 'selected' : ''}"
        style="--pct:${pct}%"
        data-index="${index}">
        <span>${opt.text}</span>
        <span class="poll-pct">${pct}%</span>
      </button>`;
  }).join('');
  document.getElementById('pollTotal').textContent = `${total} صوت · اختر بحكمة`;
}

function vote(index) {
  if (pollData.voted) return;
  pollData.options[index].votes++;
  pollData.voted = true;
  pollData.selected = index;
  savePollState();
  renderPoll();
}

function addShoutout() {
  const nameInput = document.getElementById('shoutName');
  const msgInput = document.getElementById('shoutMsg');
  const name = nameInput.value.trim();
  const msg = msgInput.value.trim();
  if (!name || !msg) return;

  const newItem = { name, text: msg, time: 'الآن' };
  shoutouts.unshift(newItem);
  saveShoutouts(shoutouts);
  renderShoutouts(shoutouts);
  nameInput.value = '';
  msgInput.value = '';
  nameInput.focus();
}

function attachPollHandlers() {
  document.getElementById('pollOptions').addEventListener('click', event => {
    const button = event.target.closest('button.poll-option');
    if (!button) return;
    const index = Number(button.dataset.index);
    vote(index);
  });
}

function init() {
  renderMembers();
  renderQuotes();
  renderGallery();

  shoutouts = loadShoutouts();
  renderShoutouts(shoutouts);

  applyPollState();
  renderPoll();

  document.getElementById('shoutSubmit').addEventListener('click', addShoutout);
  attachPollHandlers();

  updateCountdown();
  setInterval(updateCountdown, 1000);
}

document.addEventListener('DOMContentLoaded', init);
